\begin{abstract}
We provide a structural resolution of long-standing nuclear electroweak
uncertainties associated with radiative corrections and two-photon exchange
(TPE). Rather than invoking phenomenological nuclear models, we show that
admissible electroweak observables obey a rigidity constraint enforced by
locality, gauge invariance, and bounded radiative information flow.
\end{abstract}

% ============================================================
\section{Introduction}

Precision tests of the electroweak sector rely on accurate control of radiative
corrections. In nuclear and neutron beta decay, two-photon exchange (TPE)
constitutes the dominant theoretical uncertainty.

% ============================================================
\section{Admissible Electroweak Observables}

\begin{definition}[Electroweak Observable]
An electroweak observable is a gauge-invariant functional
\[
\Obs = \Obs[J^\mu_{\mathrm{em}}, J^\mu_{\mathrm{weak}}],
\]
defined on asymptotic nuclear states.
\end{definition}

% ============================================================
\section{Radiative Rigidity}

\begin{theorem}[Radiative Rigidity]
Let $\Obs$ be an admissible electroweak observable. Then all TPE contributions to
$\Obs$ lie in a finite-dimensional functional subspace
\[
\Obs_{\mathrm{TPE}} \in \Span\{\mathcal{B}_1,\ldots,\mathcal{B}_k\},
\]
where the basis is determined solely by conserved charges and currents.
\end{theorem}

\begin{proof}
Gauge invariance restricts the tensor structure of current correlators.
Locality bounds their support, and bounded radiative information flow prevents
propagation of independent high-dimensional nuclear data. Hence admissible
corrections close on a finite symmetry-determined basis.
\end{proof}

% ============================================================
\section{Conclusion}

Two-photon exchange corrections are structurally constrained. Radiative rigidity
eliminates model-dependent ambiguity and yields stable electroweak predictions.
